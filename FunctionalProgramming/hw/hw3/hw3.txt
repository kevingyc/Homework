(defun month-num (m y)(case m
(1 0)
(2 31)
(3 (if (leap? y) 60 59)
(4 (if (leap? y) 91 90)
(5 (if (leap? y) 121 120)
(6 (if (leap? y) 152 151)
(7 (if (leap? y) 182 181)
(8 (if (leap? y) 213 212)
(9 (if (leap? y) 244 243)
(10 (if (leap? y) 274 273)
(11 (if (leap? y) 305 304)
(12 (if (leap? y) 335 334)
(otherwise "unknown month")))
